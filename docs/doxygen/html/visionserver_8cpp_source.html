124bool VisionServer::updateFromConfig(const char* file) {
125 std::error_code ec;
126 wpi::raw_fd_istream is(file, ec);
127 if (ec) {
128 wpi::errs() << "Could not open '" << file << "': " << ec.message() << newline;
129 return false;
130 }
131 wpi::json j;
132 try { j = wpi::json::parse(is); }
133 catch (const wpi::json::parse_error& e) {
134 wpi::errs() << "Config error in " << file << ": byte " << e.byte << ": " << e.what() << newline;
135 return false;
136 }
137 if (!j.is_object()) {
138 wpi::errs() << "Config error in " << file << ": must be JSON object\n";
139 return false;
140 }
141 if(j.count("ntmode") != 0 && !nt::NetworkTableInstance::GetDefault().IsConnected()) {
142 try {
143 std::string str = j.at("ntmode").get<std::string>();
144 wpi::StringRef s(str);
145 if(s.equals_lower("client")) {
146 wpi::outs() << "Setting up NetworkTables in CLIENT mode\n";
147 try {
148 nt::NetworkTableInstance::GetDefault().StartClientTeam(j.at("team").get<unsigned int>());
149 nt::NetworkTableInstance::GetDefault().StartDSClient();
150 }
151 catch (const wpi::json::exception& e) {
152 wpi::errs() << "Config error in " << file << ": could not read team number: " << e.what() << newline;
153 return false;
154 }
155 } else if (s.equals_lower("server")) {
156 wpi::outs() << "Setting up NetworkTables in SERVER mode\n";
157 nt::NetworkTableInstance::GetDefault().StartServer();
158 } else {
159 wpi::errs() << "Config error in " << file << ": could not understand ntmode value '" << str << "'\n";
160 }
161 } catch (const wpi::json::exception& e) {
162 wpi::errs() << "Config error in " << file << ": coud not read ntmode: " << e.what() << newline;
163 }
164 }
165 if(j.count("cameras") < 1) {
166 wpi::errs() << "Config error in " << file << ": no camera configs found, this program requires cameras to function\n";
167 } else {
168 std::vector<VisionCamera> backup(std::move(this->cameras));
169 this->cameras.clear();
170 try {
171 for(const wpi::json& camera : j.at("cameras")) {
172 if(camera.count("calibration") > 0 && j.count("calibrations") > 0) {
173 wpi::json calibration;
174 try {
175 calibration = j.at("calibrations").at(camera.at("calibration").get<std::string>());
176 this->cameras.emplace_back(camera, calibration);
177 } catch (const wpi::json::exception& e) {
178 wpi::errs() << "Config error in " << file << ": failed to get configuration object: " << e.what() << newline; // print out more info if needed
179 this->cameras.emplace_back(camera);
180 }
181 } else { this->cameras.emplace_back(camera); }
182 this->cameras.back().setNetworkBase(this->vision);
183 this->cameras.back().setNetworkAdjustable();
184 }
185 this->vision->PutNumber("Cameras Available", this->cameras.size());
186 }
187 catch (const wpi::json::exception& e) {
188 wpi::errs() << "Config error in " << file << ": could not read cameras: " << e.what() << newline;
189 this->cameras = std::move(backup);
190 }
191 }
192 return true;
193}

232void VisionServer::visionWorker(VisionServer& server, int8_t quality) {
233
234 server.stream.SetCompression(quality); // this should call the VisionServer method directly instead of from MjpegServer
235 server.stream.SetDefaultCompression(quality);
236 server.vision->PutBoolean("Show Statistics", false);
237
238 cv::Mat frame(server.getCurrentResolution(), CV_8UC3);
239 while(server.runloop) {
240 server.source.GrabFrame(frame);
241 server.beg = server.end = CHRONO::high_resolution_clock::now();
242 server.total_frames++;
243
244 server.total_time = CHRONO::duration<double>(server.end - server.start).count();
245 server.frame_time = CHRONO::duration<double>(server.end - server.beg).count();
246 server.loop_time = CHRONO::duration<double>(server.end - server.last).count();
247 server.last = server.end;
248
249 server.active_target.update(server.loop_time);
250
251 server.fps = 1.f/server.loop_time;
252 if((int)server.total_time > (int)server.sec1_time) {
253 server.fps_1s = ((server.total_frames - server.sec1_frames) / (server.total_time - server.sec1_time));
254 server.sec1_time = server.total_time;
255 server.sec1_frames = server.total_frames;
256 }
257
258 if(server.vision->GetEntry("Show Statistics").GetBoolean(false)) {
259 server.putStats(frame);
260 }
261 server.updateStats();
262 server.output.PutFrame(frame);
263 }
264 server.vision->Delete("Show Statistics");
265 server.runloop = true;
266}
