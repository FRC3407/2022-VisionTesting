7static void splitFileParts(const char* path, char** filename, char** fileextension, char** fileparents)
8{
9 const char* slashPos = strrchr(path, FILE_PATH_SEP);
10 if (!slashPos) {
11 *fileparents = malloc(1);
12 strcpy(*fileparents, "");
13
14 const char* dotPos = strrchr(path, '.');
15 if (!dotPos) { // no extension
16 *fileextension = malloc(1);
17 strcpy(*fileextension, "");
18 *filename = malloc(strlen(path));
19 strcpy(*filename, path);
20 }
21 else {
22 *fileextension = malloc(strlen(dotPos) + 1);
23 strcpy(*fileextension, dotPos);
24 size_t bytes = dotPos - path; // do not add the dot
25 *filename = malloc(bytes + 1);
26 strncpy(*filename, path, bytes);
27 (*filename)[bytes] = '\0';
28 }
29 }
30 else {
31 size_t bytes = slashPos - path + 1;
32 *fileparents = malloc(bytes + 1);
33 strncpy(*fileparents, path, bytes);
34 (*fileparents)[bytes] = '\0';
35
36 const char* dotPos = strrchr(path, '.');
37 if (dotPos < slashPos || dotPos == NULL) { // not an extension
38 *fileextension = malloc(1);
39 strcpy(*fileextension, "");
40 *filename = malloc(strlen(slashPos));
41 strcpy(*filename, slashPos + 1);
42 }
43 else {
44 *fileextension = malloc(strlen(dotPos));
45 strcpy(*fileextension, dotPos);
46 size_t bytes = dotPos - slashPos - 1;
47 *filename = malloc(bytes + 1);
48 strncpy(*filename, slashPos + 1, bytes);
49 (*filename)[bytes] = '\0';
50 }
51
52 }
53}

55static bool splitMimeTypeParts(const char* target, char** firstPart, char** lastPart, char** trailingPart)
56{
57 const size_t MAX_PART_SIZE = 64;
58 char* mimeFirstPart = malloc(MAX_PART_SIZE), * mimeLastPart = malloc(MAX_PART_SIZE),
59 * trailingDetails = malloc(MAX_PART_SIZE * 2);
60 const char* slashPosition = strchr(target, '/');
61 if (!slashPosition) return false;
62 const char* colonPosition = strchr(target, ';');
63 if (!colonPosition) colonPosition = target + strlen(target);
64
65 size_t firstPartLen = slashPosition - target;
66 assert(firstPartLen < MAX_PART_SIZE);
67 strncpy(mimeFirstPart, target, firstPartLen);
68 mimeFirstPart[firstPartLen] = '\0';
69
70 size_t lastPartLen = colonPosition - slashPosition;
71 assert(lastPartLen < MAX_PART_SIZE);
72 strncpy(mimeLastPart, slashPosition + 1, lastPartLen - 1);
73 mimeLastPart[lastPartLen - 1] = '\0';
74
75 if (*colonPosition) {
76 size_t trailingLen = target + strlen(target) - colonPosition;
77 assert(trailingLen < MAX_PART_SIZE * 2);
78 strncpy(trailingDetails, colonPosition + 1, trailingLen - 1);
79 trailingDetails[trailingLen - 1] = '\0';
80 }
81 else {
82 trailingDetails[0] = '\0';
83 }
84 *firstPart = mimeFirstPart;
85 *lastPart = mimeLastPart;
86 *trailingPart = trailingDetails;
87
88 return true;
89}

103
104 free(targetFirstPart); free(targetLastPart); free(targetDetailsPart);
105 free(mimeFirstPart); free(mimeLastPart); free(mimeDetailsPart);

162static const char* guessFileEncoding(const char* url)
163{
164 FILE* pFile = fopen(url, "rb");
165 if (!pFile) return NULL;
166
167 // A is just a place holder
168 unsigned char firstByte = fgetc(pFile);
169 unsigned char secondByte = !feof(pFile) ? fgetc(pFile) : 'A';
170 unsigned char thirdByte = !feof(pFile) ? fgetc(pFile) : 'A';
171 unsigned char fourthByte = !feof(pFile) ? fgetc(pFile) : 'A';
172
173 if (firstByte == 0xEF && secondByte == 0xBB && thirdByte == 0xBF) {
174 return "UTF-8";
175 }
176 else if (firstByte == 0xFF && secondByte == 0xFE) {
177 return "UTF-16LE";
178 }
179 else if (firstByte == 0xFE && secondByte == 0xFF) {
180 return "UTF-16BE";
181 }
182 else if (firstByte == 0xFF && secondByte == 0xFE && thirdByte == 0x0 && fourthByte == 0x0) {
183 return "UTF-32LE";
184 }
185 else if (firstByte == 0x00 && secondByte == 0x00 && thirdByte == 0xFE && fourthByte == 0xFF) {
186 return "UTF-32BE";
187 }
188 else return NULL;
189}

213const char** getMegaMimeExtensions(const char* pMimeType)
214{
215 const char* ext;
216 size_t number = 5, pos = 0;
217 const char** extensions = calloc(sizeof(char*), number);
218
219 searchThroughMimes(NULL, NULL, NULL, NULL, true, true); // reset
220 while (searchThroughMimes(pMimeType, &ext, NULL, NULL, true, false)) {
221 if (pos == number - 1) {
222 number += 5;
223 extensions = (const char**)realloc(extensions, (sizeof(char*)) * (number));
224 }
225 extensions[pos++] = ext;
226 }
227 extensions[pos] = NULL;
228 if (!pos) {
229 free(extensions);
230 return NULL;
231 }
232 else return extensions;

236bool isTextFile(const char* url)
237{
238 if (isNotReadableFile(url))
239 {
240 return false;
241 }
242
243 const char* zEnc = guessFileEncoding(url);
244 if (!zEnc) zEnc = "UTF-8";
245
246
247 if (strcmp(zEnc, "UTF-8") == 0) {
248 FILE* pFile = fopen(url, "r");
249 if (!pFile) return false;
250
251 char ch = fgetc(pFile);
252 while (!feof(pFile) && ch != EOF) {
253 if (iscntrl(ch) && !isspace(ch)) { fclose(pFile); return 0; }
254 ch = fgetc(pFile);
255 }
256 fclose(pFile);
257 return true;
258 }
259 else {
260 FILE* pFile = fopen(url, "r,ccs=UNICODE");
261 if (!pFile) return false;
262
263 for (wint_t ch = fgetwc(pFile); ch != WEOF && !feof(pFile); ch = fgetwc(pFile)) {
264 if (iswcntrl(ch) && !iswspace(ch)) { fclose(pFile); return 0; }
265 }
266 fclose(pFile);
267 return true;
268 }
269 return false;
270}
